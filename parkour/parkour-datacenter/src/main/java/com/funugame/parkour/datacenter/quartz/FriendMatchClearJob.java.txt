package com.funugame.parkour.datacenter.quartz;

import org.quartz.Job;
import org.quartz.JobExecutionContext;

import com.funugame.parkour.dao.UserFriendMapper;
import com.funugame.parkour.dao.UserFriendMatchCountMapper;

public class FriendMatchClearJob implements Job {
    @Override
    public void execute(JobExecutionContext context) {
	try {

	    UserFriendMapper userFriendMapper = (UserFriendMapper) context.getScheduler().getContext().get("userFriendMapper");
	    UserFriendMatchCountMapper userFriendMatchCountMapper = (UserFriendMatchCountMapper) context.getScheduler().getContext().get("userFriendMatchCountMapper");
	    userFriendMapper.clearAllResult();
	    userFriendMatchCountMapper.clearAll();
	} catch (Exception e) {
	    e.printStackTrace();
	}
    }
}
