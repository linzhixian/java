package com.funugame.parkour.datacenter.quartz;

import org.quartz.Job;
import org.quartz.JobExecutionContext;

import com.funugame.parkour.dao.UserVaryMapper;

public class ClearCupLevelJob implements Job {
    @Override
    public void execute(JobExecutionContext context) {
	try {

	    UserVaryMapper userVaryMapper = (UserVaryMapper) context.getScheduler().getContext().get("userVaryMapper");
	    userVaryMapper.clearCupLevel();
	} catch (Exception e) {
	    e.printStackTrace();
	}
    }
}
