import java.io.File;
import java.io.IOException;
import java.util.HashMap;

import com.lzx.framework.utils.file.CharSetConvertor;

public class TestLoad {

    public static void main(String[] args) throws IOException {
	// loadClientNeedFiles();
	HashMap<String, Object> map = new HashMap<String, Object>();
	map.put("level",new Integer(4));
	Integer level=new Integer(4);
	if(map.get("level")!=level) {
	    System.out.println("not equals");
	}
    }

    private static void loadClientNeedFiles() {
	byte[] bytes = loadConfigResource("clientNeedFiles.txt");
	String content = CharSetConvertor.convertBytes(bytes, "clientNeedFiles.txt");
	String as1[] = org.apache.commons.lang.StringUtils.splitByWholeSeparator(content, "\r\n");
	for (String s : as1) {
	    System.out.println("f:[" + s.trim() + "]");
	}
    }

    private static byte[] loadConfigResource(String fileName) {
	String confPath = Thread.currentThread().getContextClassLoader().getResource("conf" + "/" + fileName).getPath();
	File f = new File(confPath);
	return CharSetConvertor.readFileToTypes(f);
    }
}
