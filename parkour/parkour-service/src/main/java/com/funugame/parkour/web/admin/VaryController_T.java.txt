package com.funugame.parkour.web.admin;

import java.util.ArrayList;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;

import com.funugame.parkour.api.IUserService;
import com.funugame.parkour.api.daoentity.UserInfo;
import com.funugame.parkour.api.daoentity.UserVary;

//TestVaryController
@Controller
@RequestMapping(value = "/vary")
@Lazy
public class VaryController_T {
    @Autowired
    private IUserService userService;

    @RequestMapping(value = "/list", method = RequestMethod.GET)
    public String list(Model model) {
	model.addAttribute(new UserVary());
	return "uservary/list";
    }

    @RequestMapping(value = "/list", method = RequestMethod.POST)
    public String list2(@Validated UserVary uservary, BindingResult br, Model model) {
	try {
	    uservary = userService.selectIdList(uservary.getUid());
	} catch (Exception e) {
	    e.printStackTrace();
	}
	List list = new ArrayList();
	list.add(uservary);
	model.addAttribute("list", list);
	return "uservary/list";
    }

    @RequestMapping(value = "/{uid}/update", method = RequestMethod.GET)
    public String update(@PathVariable int uid, Model model) {
	model.addAttribute(userService.selectIdList(uid));
	return "uservary/update";
    }

    @RequestMapping(value = "/{uid}/update", method = RequestMethod.POST)
    public String update(@PathVariable int uid, @Validated UserVary uservary, BindingResult br, Model model) {
	try {
	    userService.updateVary(uservary);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/vary/list";
    }

}
