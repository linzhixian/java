package com.funugame.parkour.web.admin;

import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;

import com.funugame.parkour.api.IUserService;
import com.funugame.parkour.api.daoentity.Notice;
import com.funugame.parkour.api.daoentity.UserVary;

@Controller
@RequestMapping(value = "/notice")
@Lazy
public class NoticeController_T {
    @Autowired
    private IUserService userService;

    @RequestMapping(value = "/list", method = RequestMethod.GET)
    public String list(Model model) {
	model.addAttribute(new Notice());
	return "notice/list";
    }

    @RequestMapping(value = "/list", method = RequestMethod.POST)
    public String list2(@Validated Notice nt, BindingResult br, Model model) {
	List<Notice> list=new ArrayList<Notice>();
	try {
	    list= userService.selectNotice(nt);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	return "notice/list";
    }

    @RequestMapping(value = "/{type}/update", method = RequestMethod.GET)
    public String update(@PathVariable int type,@Validated Notice nt, Model model) {
	nt.setType(type);
	model.addAttribute(userService.selectNotice(nt));
	return "notice/update";
    }

    @RequestMapping(value = "/{type}/update", method = RequestMethod.POST)
    public String update(@Validated Notice nt, BindingResult br, Model model) {
	try {
	    userService.updateNoticeList(nt);
	    ;
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/notice/list";
    }

}
