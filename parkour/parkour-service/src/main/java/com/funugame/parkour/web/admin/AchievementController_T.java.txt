package com.funugame.parkour.web.admin;

import java.util.ArrayList;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;

import com.funugame.parkour.api.ISyncService;
import com.funugame.parkour.api.IUserService;
import com.funugame.parkour.api.daoentity.UserAchievement;

@Controller
@RequestMapping(value = "/achievement")
@Lazy
public class AchievementController_T {
    @Autowired
    private IUserService userService;
    @Autowired
    private ISyncService syncService;

    @RequestMapping(value = "/list", method = RequestMethod.GET)
    public String list(Model model) {
	model.addAttribute(new UserAchievement());
	return "achievements/select";
    }

    @RequestMapping(value = "/list", method = RequestMethod.POST)
    public String list2(@Validated UserAchievement userAchievement, BindingResult br, Model model) {
	List list = new ArrayList();
	try {
	    list = userService.selectAchievementList(userAchievement);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	return "achievements/list";
    }

    @RequestMapping(value = "/add", method = RequestMethod.GET)
    public String add(Model model) {
	model.addAttribute(new UserAchievement());
	return "achievements/add";
    }

    @RequestMapping(value = "/add", method = RequestMethod.POST)
    public String add(@Validated UserAchievement userAchievement, BindingResult br, Model model) {
	try {
	    userAchievement.setMark("insert");
	    List ls = new ArrayList();
	    ls.add(userAchievement);
	    Integer uid=userAchievement.getUid();
	    syncService.syncAchievement(uid,ls);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	List list = new ArrayList();
	try {
	    list = userService.selectAchievementList(userAchievement);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	model.addAttribute(userAchievement);
	return "achievements/list";
    }

    @RequestMapping(value = "{uid},{id}/update", method = RequestMethod.GET)
    public String update(@PathVariable int uid, @PathVariable int id, @Validated UserAchievement userAchievement, Model model) {
	userAchievement.setUid(uid);
	userAchievement.setId(id);
	List list = new ArrayList();
	try {
	    list = userService.selectAchievementList(userAchievement);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	userAchievement = (UserAchievement) list.get(0);
	model.addAttribute(userAchievement);
	return "achievements/update";
    }

    @RequestMapping(value = "{uid},{id}/update", method = RequestMethod.POST)
    public String update2(@PathVariable int uid, @PathVariable int id, @Validated UserAchievement userAchievement, BindingResult br, Model model) {
	try {
	    userAchievement.setMark("update");
	    List ls = new ArrayList();
	    ls.add(userAchievement);
	    uid=userAchievement.getUid();
	    syncService.syncAchievement(uid,ls);
	    model.addAttribute("list", ls);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/achievement/list";
    }

    @RequestMapping(value = "/{uid},{id}/delete", method = RequestMethod.GET)
    public String delete(@PathVariable int uid, @PathVariable int id) {
	UserAchievement userAchievement = new UserAchievement();
	userAchievement.setMark("delete");
	userAchievement.setUid(uid);
	userAchievement.setId(id);
	List list = new ArrayList();
	try {
	    list.add(userAchievement);
	    uid=userAchievement.getUid();
	    syncService.syncAchievement(uid,list);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/achievement/list";
    }
}
