package com.funugame.parkour.web.admin;

import java.util.ArrayList;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;

import com.funugame.parkour.api.ISyncService;
import com.funugame.parkour.api.IUserService;
import com.funugame.parkour.api.daoentity.UserProp;

@Controller
@RequestMapping(value = "/prop")
@Lazy
public class PropController_T {
    @Autowired
    private IUserService userService;
    @Autowired
    private ISyncService syncService;

    @RequestMapping(value = "/list", method = RequestMethod.GET)
    public String list(Model model) {
	model.addAttribute(new UserProp());
	return "props/select";
    }

    @RequestMapping(value = "/list", method = RequestMethod.POST)
    public String list2(@Validated UserProp userProp, BindingResult br, Model model) {
	List list = new ArrayList();
	try {
	    list = userService.selectPropList(userProp);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	return "props/list";
    }

    @RequestMapping(value = "/add", method = RequestMethod.GET)
    public String add(Model model) {
	model.addAttribute(new UserProp());
	return "props/add";
    }

    @RequestMapping(value = "/add", method = RequestMethod.POST)
    public String add(@Validated UserProp userProp, BindingResult br, Model model) {
	try {
	    userProp.setMark("insert");
	    List ls = new ArrayList();
	    ls.add(userProp);
	    Integer uid=userProp.getUid();
	    syncService.syncProp(uid,ls);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	List list = new ArrayList();
	try {
	    list = userService.selectPropList(userProp);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	model.addAttribute(userProp);
	return "props/list";
    }

    @RequestMapping(value = "{uid},{id}/update", method = RequestMethod.GET)
    public String update(@PathVariable int uid, @PathVariable int id, @Validated UserProp userProp, Model model) {
	userProp.setUid(uid);
	userProp.setId(id);
	List list = new ArrayList();
	try {
	    list = userService.selectPropList(userProp);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	userProp = (UserProp) list.get(0);
	model.addAttribute(userProp);
	return "props/update";
    }

    @RequestMapping(value = "{uid},{id}/update", method = RequestMethod.POST)
    public String update2(@PathVariable int uid, @PathVariable int id, @Validated UserProp userProp, BindingResult br, Model model) {
	try {
	    userProp.setMark("update");
	    List ls = new ArrayList();
	    ls.add(userProp);
	    uid=userProp.getUid();
	    syncService.syncProp(uid,ls);
	    model.addAttribute("list", ls);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/prop/list";
    }

    @RequestMapping(value = "/{uid},{id}/delete", method = RequestMethod.GET)
    public String delete(@PathVariable int uid, @PathVariable int id) {
	UserProp userProp = new UserProp();
	userProp.setMark("delete");
	userProp.setUid(uid);
	userProp.setId(id);
	List list = new ArrayList();
	try {
	    list.add(userProp);
	    uid=userProp.getUid();
	    syncService.syncProp(uid,list);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/prop/list";
    }
}
