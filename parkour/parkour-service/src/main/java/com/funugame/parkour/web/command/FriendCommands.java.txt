package com.funugame.parkour.web.command;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Service;

import com.funugame.parkour.api.ISyncService;
import com.funugame.parkour.api.jsonbeans.AddFriend;
import com.lzx.framework.web.springjson.annotation.WebAction;
import com.lzx.framework.web.springjson.annotation.WebCommand;
import com.lzx.framework.web.springjson.beans.ResponseBase;

@WebCommand
@Service
@Lazy
public class FriendCommands {

    @Autowired
    private ISyncService syncService;
    
    @WebAction(name = "addFriend")
    public ResponseBase addFriend(AddFriend request) {
	int result=syncService.addFriend(request.getUid(),request.getFuid());
	ResponseBase rb=new ResponseBase();
	rb.setCode(result);
	return rb;
    }
    
}
