package com.funugame.parkour.web.admin;

import java.util.ArrayList;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.validation.BindingResult;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.w3c.dom.ls.LSException;

import com.funugame.parkour.api.ISyncService;
import com.funugame.parkour.api.IUserService;
import com.funugame.parkour.api.daoentity.UserInfo;
import com.funugame.parkour.api.daoentity.UserPet;

@Controller
@RequestMapping(value = "/pet")
@Lazy
public class PetController_T {
    @Autowired
    private IUserService userService;
    @Autowired
    private ISyncService syncService;

    @RequestMapping(value = "/list", method = RequestMethod.GET)
    public String list(Model model) {
	model.addAttribute(new UserPet());
	return "pets/select";
    }

    @RequestMapping(value = "/list", method = RequestMethod.POST)
    public String list2(@Validated UserPet userpet, BindingResult br, Model model) {
	List list = new ArrayList();
	try {
	    list = userService.selectPetList(userpet);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	return "pets/list";
    }

    @RequestMapping(value = "/add", method = RequestMethod.GET)
    public String add(Model model) {
	model.addAttribute(new UserPet());
	return "pets/add";
    }

    @RequestMapping(value = "/add", method = RequestMethod.POST)
    public String add(@Validated UserPet userpet, BindingResult br, Model model) {
	try {
	    userpet.setMark("insert");
	    List ls=new ArrayList();
	    ls.add(userpet);
	    Integer uid=userpet.getUid();
	    syncService.syncPet(uid,ls);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	List list = new ArrayList();
	try {
	    list = userService.selectPetList(userpet);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	model.addAttribute("list", list);
	model.addAttribute(userpet);
	return "pets/list";
    }

    @RequestMapping(value = "{uid},{id}/update", method = RequestMethod.GET)
    public String update(@PathVariable int uid,@PathVariable int id,@Validated UserPet userpet,Model model) {
	userpet.setUid(uid);
	userpet.setId(id);
	List list = new ArrayList();
	try {
	    list = userService.selectPetList(userpet);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	userpet=(UserPet) list.get(0);
	model.addAttribute(userpet);
	return "pets/update";
    }

    @RequestMapping(value = "{uid},{id}/update", method = RequestMethod.POST)
    public String update2(@PathVariable int uid,@PathVariable int id,@Validated UserPet userpet, BindingResult br, Model model) {
	try {
	    userpet.setMark("update");
	    List ls=new ArrayList();
	    ls.add(userpet);
	    uid=userpet.getUid();
	    syncService.syncPet(uid,ls);
	    model.addAttribute("list",ls);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/pet/list";
    }

    @RequestMapping(value = "/{uid},{id}/delete", method = RequestMethod.GET)
    public String delete(@PathVariable int uid,@PathVariable int id) {
	UserPet userpet = new UserPet();
	userpet.setMark("delete");
	userpet.setUid(uid);
	userpet.setId(id);
	List list = new ArrayList();
	try {
	    list.add(userpet);
	    uid=userpet.getUid();
	    syncService.syncPet(uid,list);
	} catch (Exception e) {
	    e.printStackTrace();
	}
	return "redirect:/pet/list";
    }
}
