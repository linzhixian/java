package com.funugame.parkour.web.command;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Service;

import com.funugame.parkour.api.IMatchService;
import com.funugame.parkour.api.daoentity.MatchUser;
import com.funugame.parkour.api.jsonbeans.UidAndChapterId;
import com.lzx.framework.web.springjson.annotation.WebAction;
import com.lzx.framework.web.springjson.annotation.WebCommand;

@WebCommand
@Service
@Lazy
public class MatchCommands {
    @Autowired
    private IMatchService matchService;

    /**
     * 获取挑战进阶用户列表 @author lzx 2014年11月20日上午11:07:29
     */
    @WebAction(name = "matchUserList")
    public List<MatchUser> matchUserList(UidAndChapterId request) {
	return matchService.getMathUserList(request.getUid(), request.getChapterid());
    }

    /**
     * 强制重新获取挑战进阶用户列表 @author lzx 2014年11月20日上午11:07:39
     */
    @WebAction(name = "reGenerateMatchUserList")
    public List<MatchUser> reGenerateMatchUserList(UidAndChapterId request) {
	return matchService.reGenerateMatchUserList(request.getUid(), request.getChapterid());
    }
}
