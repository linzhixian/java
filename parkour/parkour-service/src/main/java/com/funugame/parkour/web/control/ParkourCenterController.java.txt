package com.funugame.parkour.web.control;

import java.util.Map;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Lazy;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestHeader;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.ResponseBody;

import com.funugame.parkour.api.IUserService;
import com.funugame.parkour.web.RequestJson;
import com.lzx.framework.utils.MD5;
import com.lzx.framework.utils.StringUtil;
import com.lzx.framework.web.springjson.beans.IRequestJson;
import com.lzx.framework.web.springjson.controller.CenterController;

@Controller
@Lazy
@RequestMapping(value = "/")
public class ParkourCenterController extends CenterController {


    @Autowired
    private IUserService userService;


    @RequestMapping(value = "/controller", method = RequestMethod.POST)
    @ResponseBody
    public Map<String, Object> controller(@RequestBody RequestJson requestJson,@RequestHeader Map<String, String> pHeader) {
	System.out.println(StringUtil.objectToString(pHeader));
	String userAgent = pHeader.get("user-agent");
	System.out.println("user-agent--------------------------"+userAgent);
	requestJson.setHeader(pHeader);
	return super.controller(requestJson);
    }
   
    @Override
    protected boolean checkSign(Integer uid, String sign, String source) {
	// String signmd5 = MD5.md5String(userService.getMd5Key(uid) + source);
	return MD5.md5String(userService.getMd5Key(uid) + source).equals(sign);
    }

    @Override
    protected Class getRequestJsonClass() {
	return RequestJson.class;
    }

}
