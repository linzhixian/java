package com.funugame.parkour.api.beans;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

/**
 * 
 * @author linzhixian 2014年10月24日下午2:16:41 客户端配置对象：用了返回给客户端的配置文件内容
 */
public class ClientConfig implements Serializable{
    private static final long serialVersionUID = 5756903367728912739L;
    private int version;
    private Map<String, ConfigFile> fileList = new HashMap<String, ConfigFile>();

    public int getVersion() {
	return version;
    }

    public void setVersion(int version) {
	this.version = version;
    }

    public Map<String, ConfigFile> getFileList() {
	return fileList;
    }

    public void addConfigFile(String fileName, ConfigFile file) {
	fileList.put(fileName, file);
    }
}
