package com.funugame.parkour.api.daoentity;

import java.io.Serializable;

import javax.management.loading.PrivateClassLoader;

import com.fasterxml.jackson.annotation.JsonIgnore;

/**
 * 碎片
 * 
 * @author linzhixian 2014年11月8日下午4:51:11
 * 
 */
public class UserPiece implements Serializable {
    private static final long serialVersionUID = 1L;

    private int uid;
    private int id;
    private int piece;
    private String mark;

    public UserPiece() {

    }

    public UserPiece(int uid, int id) {
	this.uid = uid;
	this.id = id;
    }
    public String getMark() {
	return mark;
    }

    public void setMark(String mark) {
	this.mark = mark;
    }

    public int getUid() {
	return uid;
    }

    public void setUid(int uid) {
	this.uid = uid;
    }

    public int getId() {
	return id;
    }

    public void setId(int id) {
	this.id = id;
    }

    public int getPiece() {
	return piece;
    }

    public void setPiece(int piece) {
	this.piece = piece;
    }
}
