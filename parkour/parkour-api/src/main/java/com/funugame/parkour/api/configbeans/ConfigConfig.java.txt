package com.funugame.parkour.api.configbeans;

import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

import com.lzx.framework.utils.collection.IntHashMap;

public class ConfigConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();
    private String key;
    private String value;
    private String desc;

    public static Map<String, BaseConfig> ALLS;

    public static String getValue(String key) {
	if (ALLS == null)
	    init();
	return ((ConfigConfig) ALLS.get(key)).getValue();
    }

    private static void init() {
	ALLS = new HashMap<String, BaseConfig>();
	Iterator<BaseConfig> iter = ALL.values().iterator();
	while (iter.hasNext()) {
	    ConfigConfig config = (ConfigConfig) iter.next();
	    ALLS.put(config.getKey(), config);
	}
    }

    public static int getIntValue(String key) {
	if (ALLS == null)
	    init();
	return Integer.parseInt(((ConfigConfig) ALLS.get(key)).getValue());
    }

    public String getKey() {
	return key;
    }

    public void setKey(String key) {
	this.key = key;
    }

    public String getValue() {
	return value;
    }

    public void setValue(String value) {
	this.value = value;
    }

    public String getDesc() {
	return desc;
    }

    public void setDesc(String desc) {
	this.desc = desc;
    }
}
