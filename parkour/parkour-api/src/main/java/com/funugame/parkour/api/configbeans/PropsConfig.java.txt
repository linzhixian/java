package com.funugame.parkour.api.configbeans;

import com.lzx.framework.utils.collection.IntHashMap;

public class PropsConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();
    private String name;
    private int type;
    private int price;
    private int buytype;
    private int unlocklevel;
    private int nextBuyAddDiamond;
    private float minBuycount;
    private float dayUseLimit;
    private int addAge;
    private int lifeRegen;
    private int speedDistance;
    private boolean relive;
    private String memo;

    public static PropsConfig getById(int id) {
	return (PropsConfig) ALL.get(id);
    }

    public String getName() {
	return name;
    }

    public void setName(String name) {
	this.name = name;
    }

    public int getType() {
	return type;
    }

    public void setType(int type) {
	this.type = type;
    }

    public float getMinBuycount() {
	return minBuycount;
    }

    public void setMinBuycount(float minBuycount) {
	this.minBuycount = minBuycount;
    }

    public float getDayUseLimit() {
	return dayUseLimit;
    }

    public void setDayUseLimit(float dayUseLimit) {
	this.dayUseLimit = dayUseLimit;
    }

    public int getAddAge() {
	return addAge;
    }

    public void setAddAge(int addAge) {
	this.addAge = addAge;
    }

    public int getLifeRegen() {
	return lifeRegen;
    }

    public void setLifeRegen(int lifeRegen) {
	this.lifeRegen = lifeRegen;
    }

    public int getSpeedDistance() {
	return speedDistance;
    }

    public void setSpeedDistance(int speedDistance) {
	this.speedDistance = speedDistance;
    }

    public boolean isRelive() {
	return relive;
    }

    public void setRelive(boolean relive) {
	this.relive = relive;
    }

    public String getMemo() {
	return memo;
    }

    public void setMemo(String memo) {
	this.memo = memo;
    }

    public int getNextBuyAddDiamond() {
	return nextBuyAddDiamond;
    }

    public void setNextBuyAddDiamond(int nextBuyAddDiamond) {
	this.nextBuyAddDiamond = nextBuyAddDiamond;
    }

    public int getPrice() {
	return price;
    }

    public void setPrice(int price) {
	this.price = price;
    }

    public int getBuytype() {
	return buytype;
    }

    public void setBuytype(int buytype) {
	this.buytype = buytype;
    }

    public int getUnlocklevel() {
	return unlocklevel;
    }

    public void setUnlocklevel(int unlocklevel) {
	this.unlocklevel = unlocklevel;
    }
}
