package com.funugame.parkour.api.configbeans;

import com.lzx.framework.utils.collection.IntHashMap;

public class RefleshMatchListConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();

    public static int[] HOURS;
    private int hour;

    public static int[] getHours() {
	if (HOURS == null) {
	    HOURS = new int[ALL.size()];
	    for (int i = 0; i < HOURS.length; i++) {
		HOURS[i] = ((RefleshMatchListConfig) ALL.get(i)).getHour();
	    }
	}
	return HOURS;
    }

    public int getHour() {
	return hour;
    }

    public void setHour(int hour) {
	this.hour = hour;
    }

}
