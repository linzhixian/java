package com.funugame.parkour.api.jsonbeans;

import java.io.Serializable;

public class Login implements Serializable {
    private static final long serialVersionUID = 1L;

    public final static int LOGIN_NORMAL = 0;
    public final static int LOGIN_QQ = 1;
    public final static int LOGIN_WX = 2;

    private Integer type;// 0：普通登录，1：QQ登录，2：微信登录
    private Integer uid;
    private String password;
    private String openid;
    private String figureurl_qq;
    private String qqname;

    private String userAgent;

    public Integer getUid() {
	return uid;
    }

    public void setUid(Integer uid) {
	this.uid = uid;
    }

    public String getPassword() {
	return password;
    }

    public void setPassword(String password) {
	this.password = password;
    }

    public String getOpenid() {
	return openid;
    }

    public void setOpenid(String openid) {
	this.openid = openid;
    }

    public String getFigureurl_qq() {
	return figureurl_qq;
    }

    public void setFigureurl_qq(String figureurl_qq) {
	this.figureurl_qq = figureurl_qq;
    }

    public String getQqname() {
	return qqname;
    }

    public void setQqname(String qqname) {
	this.qqname = qqname;
    }

    public Integer getType() {
	return type;
    }

    public void setType(Integer type) {
	this.type = type;
    }

    public String getUserAgent() {
        return userAgent;
    }

    public void setUserAgent(String userAgent) {
        this.userAgent = userAgent;
    }
}
