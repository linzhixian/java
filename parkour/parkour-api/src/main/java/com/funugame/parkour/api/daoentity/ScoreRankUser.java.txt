package com.funugame.parkour.api.daoentity;

import java.io.Serializable;

public class ScoreRankUser implements Serializable {

    private static final long serialVersionUID = 1L;

    private int uid;
    private int cid;
    private String rolename;
    private String figureurl_qq;
    private int score;
    private int dist;
    private int age;
    private int cup;
    private int cuplevel;
    private int result;
    
    public int getUid() {
	return uid;
    }

    public void setUid(int uid) {
	this.uid = uid;
    }


    public int getScore() {
	return score;
    }

    public void setScore(int score) {
	this.score = score;
    }

    public int getAge() {
	return age;
    }

    public void setAge(int age) {
	this.age = age;
    }

    public int getCup() {
	return cup;
    }

    public void setCup(int cup) {
	this.cup = cup;
    }

    public int getCuplevel() {
	return cuplevel;
    }

    public void setCuplevel(int cuplevel) {
	this.cuplevel = cuplevel;
    }

    public String getRolename() {
        return rolename;
    }

    public void setRolename(String rolename) {
        this.rolename = rolename;
    }

    public int getResult() {
        return result;
    }

    public void setResult(int result) {
        this.result = result;
    }

    public int getDist() {
        return dist;
    }

    public void setDist(int dist) {
        this.dist = dist;
    }

    public int getCid() {
        return cid;
    }

    public void setCid(int cid) {
        this.cid = cid;
    }

    public String getFigureurl_qq() {
        return figureurl_qq;
    }

    public void setFigureurl_qq(String figureurl_qq) {
        this.figureurl_qq = figureurl_qq;
    }
}
