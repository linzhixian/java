package com.funugame.parkour.api.jsonbeans;

import java.io.Serializable;

import com.lzx.framework.web.springjson.beans.IUidSupport;

public class GameResult implements Serializable,IUidSupport {

    private static final long serialVersionUID = 1;
    private Integer type;// 0普通比赛 1挑战进阶 2 好友挑战
    private Integer uid;
    private Integer muid;// 挑战用户id，type=1、2时才有该值
    private Integer result;// 0平，1赢，2输； type=1、2时该值有效
    private Integer chapterid;
    private Integer distance;
    private Integer exp;
    private Integer score;
    private Integer gold;

    private Integer piece;// 碎片
    private Integer bone;// 骨头

    private int bosscount;


    public Integer getType() {
	return type;
    }

    public void setType(Integer type) {
	this.type = type;
    }

    public Integer getUid() {
	return uid;
    }

    public void setUid(Integer uid) {
	this.uid = uid;
    }

    public Integer getMuid() {
	return muid;
    }

    public void setMuid(Integer muid) {
	this.muid = muid;
    }

    public Integer getChapterid() {
	return chapterid;
    }

    public void setChapterid(Integer chapterid) {
	this.chapterid = chapterid;
    }

    public Integer getDistance() {
	return distance;
    }

    public void setDistance(Integer distance) {
	this.distance = distance;
    }

    public Integer getExp() {
	return exp;
    }

    public void setExp(Integer exp) {
	this.exp = exp;
    }

    public Integer getScore() {
	return score;
    }

    public void setScore(Integer score) {
	this.score = score;
    }

    public Integer getGold() {
	return gold;
    }

    public void setGold(Integer gold) {
	this.gold = gold;
    }

    public Integer getPiece() {
	return piece;
    }

    public void setPiece(Integer piece) {
	this.piece = piece;
    }

    public Integer getBone() {
	return bone;
    }

    public void setBone(Integer bone) {
	this.bone = bone;
    }

    public Integer getResult() {
        return result;
    }

    public void setResult(Integer result) {
        this.result = result;
    }

    public int getBosscount() {
        return bosscount;
    }

    public void setBosscount(int bosscount) {
        this.bosscount = bosscount;
    }

}
