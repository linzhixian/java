package com.funugame.parkour.api.configbeans;

import com.lzx.framework.utils.collection.IntHashMap;

public class AchievementConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();
    public static final int TYPE_RUN = 1;
    public static final int TYPE_CHALLENGE = 2;
    public static final int TYPE_RANK = 3;
    public static final int TYPE_GOLD = 4;
    public static final int TYPE_BOSS = 5;

    private String name;
    private int type;
    private int count;
    private int award;
    private String awardType;
    
    public static AchievementConfig getById(int id) {
	return (AchievementConfig) ALL.get(id);
    }

    public String getName() {
	return name;
    }

    public void setName(String name) {
	this.name = name;
    }

    public int getType() {
	return type;
    }

    public void setType(int type) {
	this.type = type;
    }

    public int getCount() {
	return count;
    }

    public void setCount(int count) {
	this.count = count;
    }

    public int getAward() {
	return award;
    }

    public void setAward(int award) {
	this.award = award;
    }

    public String getAwardType() {
        return awardType;
    }

    public void setAwardType(String awardType) {
        this.awardType = awardType;
    }

}
