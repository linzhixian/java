package com.funugame.parkour.api.jsonbeans;

import java.util.List;

import com.funugame.parkour.api.daoentity.User;
import com.funugame.parkour.api.daoentity.UserAchievement;
import com.funugame.parkour.api.daoentity.UserPet;
import com.funugame.parkour.api.daoentity.UserPiece;
import com.funugame.parkour.api.daoentity.UserProp;
import com.funugame.parkour.api.daoentity.UserTech;
import com.lzx.framework.web.springjson.beans.ResponseBase;

public class LoginResult extends ResponseBase {
    private static final long serialVersionUID = 1L;
    private boolean isNew;
    private String md5key;
    private User user;
    private long stime;
    private List<UserPet> pets;
    private List<UserTech> techs;
    private List<UserAchievement> achievements;
    private List<UserProp> prop;
    private List<UserPiece> piece;

    public boolean isNew() {
	return isNew;
    }

    public void setNew(boolean isNew) {
	this.isNew = isNew;
    }

    public String getMd5key() {
	return md5key;
    }

    public void setMd5key(String md5key) {
	this.md5key = md5key;
    }

    public User getUser() {
	return user;
    }

    public void setUser(User user) {
	this.user = user;
    }

    public List<UserPet> getPets() {
	return pets;
    }

    public void setPets(List<UserPet> pets) {
	this.pets = pets;
    }

    public List<UserTech> getTechs() {
	return techs;
    }

    public void setTechs(List<UserTech> techs) {
	this.techs = techs;
    }

    public List<UserAchievement> getAchievements() {
	return achievements;
    }

    public void setAchievements(List<UserAchievement> achievements) {
	this.achievements = achievements;
    }

    public List<UserProp> getProp() {
	return prop;
    }

    public void setProp(List<UserProp> prop) {
	this.prop = prop;
    }

    public List<UserPiece> getPiece() {
	return piece;
    }

    public void setPiece(List<UserPiece> piece) {
	this.piece = piece;
    }



    public long getStime() {
        return stime;
    }

    public void setStime(long stime) {
        this.stime = stime;
    }

}
