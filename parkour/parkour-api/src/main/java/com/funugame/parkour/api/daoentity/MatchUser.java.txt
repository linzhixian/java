package com.funugame.parkour.api.daoentity;

import java.io.Serializable;
import java.util.Date;

public class MatchUser implements Serializable {

    private static final long serialVersionUID = 1L;

    private int uid;    
    private int cid;
    private String rolename;
    private String figureurl_qq;
    private int score;
    private int dist;
    private int age;
    private int cuplevel;
    //竞赛结果：0没有结果，1赢，2输
    private int result;

    private Date updatetime;
    
    public int getUid() {
	return uid;
    }

    public void setUid(int uid) {
	this.uid = uid;
    }

    public String getRolename() {
	return rolename;
    }

    public void setRolename(String rolename) {
	this.rolename = rolename;
    }

    public int getScore() {
	return score;
    }

    public void setScore(int score) {
	this.score = score;
    }

    public int getAge() {
	return age;
    }

    public void setAge(int age) {
	this.age = age;
    }

    public int getCuplevel() {
	return cuplevel;
    }

    public void setCuplevel(int cuplevel) {
	this.cuplevel = cuplevel;
    }

    public int getResult() {
        return result;
    }

    public void setResult(int result) {
        this.result = result;
    }

    public Date getUpdatetime() {
        return updatetime;
    }

    public void setUpdatetime(Date updatetime) {
        this.updatetime = updatetime;
    }

    public int getDist() {
        return dist;
    }

    public void setDist(int dist) {
        this.dist = dist;
    }

    public int getCid() {
        return cid;
    }

    public void setCid(int cid) {
        this.cid = cid;
    }

    public String getFigureurl_qq() {
        return figureurl_qq;
    }

    public void setFigureurl_qq(String figureurl_qq) {
        this.figureurl_qq = figureurl_qq;
    }

}
