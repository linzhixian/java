package com.funugame.parkour.api.configbeans;

import com.lzx.framework.utils.collection.IntHashMap;

public class TechConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();

    private String name;

    private int propId;
    private int tech;
    private int initLevel;
    private int maxLevel;
    private int cup;
    private int gold;
    private String propDesc;
    private String gradeDesc;

    public static TechConfig getById(int id) {
	return (TechConfig) ALL.get(id);
    }

    public String getName() {
	return name;
    }

    public void setName(String name) {
	this.name = name;
    }

    public int getMaxLevel() {
	return maxLevel;
    }

    public void setMaxLevel(int maxLevel) {
	this.maxLevel = maxLevel;
    }

    public int getCup() {
	return cup;
    }

    public void setCup(int cup) {
	this.cup = cup;
    }

    public int getInitLevel() {
	return initLevel;
    }

    public void setInitLevel(int initLevel) {
	this.initLevel = initLevel;
    }

    public int getGold() {
	return gold;
    }

    public void setGold(int gold) {
	this.gold = gold;
    }

    public int getPropId() {
	return propId;
    }

    public void setPropId(int propId) {
	this.propId = propId;
    }

    public int getTech() {
	return tech;
    }

    public void setTech(int tech) {
	this.tech = tech;
    }

    public String getPropDesc() {
	return propDesc;
    }

    public void setPropDesc(String propDesc) {
	this.propDesc = propDesc;
    }

    public String getGradeDesc() {
	return gradeDesc;
    }

    public void setGradeDesc(String gradeDesc) {
	this.gradeDesc = gradeDesc;
    }
}
