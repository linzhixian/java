package com.funugame.parkour.api.configbeans;

import java.util.Iterator;

import com.lzx.framework.utils.collection.IntHashMap;

public class DeviceTypeConfig extends BaseConfig {

    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();

    private String name;

    public static final int UNKONW = -1;

    public static Integer getTypeByName(String name) {
	Iterator<Integer> keys = ALL.keySet().iterator();
	while (keys.hasNext()) {
	    Integer id = keys.next();
	    DeviceTypeConfig config = (DeviceTypeConfig) ALL.get(id);
	    if (name.startsWith(config.getName())) {
		return id;
	    }
	}
	return UNKONW;
    }

    public String getName() {
	return name;
    }

    public void setName(String name) {
	this.name = name;
    }
}
