package com.funugame.parkour.api.configbeans;

import com.lzx.framework.utils.collection.IntHashMap;

public class ChapterConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();

    private String name;

    public static String getNameById(int id) {
	ChapterConfig c = (ChapterConfig) ALL.get(id);
	if (c != null)
	    return c.getName();
	return "notFound:" + id;
    }

    public String getName() {
	return name;
    }

    public void setName(String name) {
	this.name = name;
    }

}
