package com.funugame.parkour.api.configbeans;

import com.lzx.framework.utils.collection.IntHashMap;

public class PetConfig extends BaseConfig {
    public static IntHashMap<BaseConfig> ALL = new IntHashMap<BaseConfig>();

    private int chapter;
    private String name;
    private int needFragment;
    private int maxLevel;
    private int price;
    private int conditionId;
    private float moneyPlusPer;
    private float moneyPlusAddLevelUp;
    private float scorePlusPer;
    private float scorePlusAddLevelUp;
    private int distance;
    private int addDistanceLevelUp;
    private int cdtime;
    private float reduceCdtimeLevelUp;
    private String desc;
    private String spell;
    private String spellDesc;

    public static PetConfig getById(int id) {
	return (PetConfig) ALL.get(id);
    }

    public String getName() {
	return name;
    }

    public void setName(String name) {
	this.name = name;
    }

    public int getConditionId() {
	return conditionId;
    }

    public void setConditionId(int conditionId) {
	this.conditionId = conditionId;
    }

    public float getMoneyPlusPer() {
	return moneyPlusPer;
    }

    public void setMoneyPlusPer(float moneyPlusPer) {
	this.moneyPlusPer = moneyPlusPer;
    }

    public float getMoneyPlusAddLevelUp() {
	return moneyPlusAddLevelUp;
    }

    public void setMoneyPlusAddLevelUp(float moneyPlusAddLevelUp) {
	this.moneyPlusAddLevelUp = moneyPlusAddLevelUp;
    }

    public float getScorePlusPer() {
	return scorePlusPer;
    }

    public void setScorePlusPer(float scorePlusPer) {
	this.scorePlusPer = scorePlusPer;
    }

    public float getScorePlusAddLevelUp() {
	return scorePlusAddLevelUp;
    }

    public void setScorePlusAddLevelUp(float scorePlusAddLevelUp) {
	this.scorePlusAddLevelUp = scorePlusAddLevelUp;
    }

    public int getDistance() {
	return distance;
    }

    public void setDistance(int distance) {
	this.distance = distance;
    }

    public int getAddDistanceLevelUp() {
	return addDistanceLevelUp;
    }

    public void setAddDistanceLevelUp(int addDistanceLevelUp) {
	this.addDistanceLevelUp = addDistanceLevelUp;
    }

    public int getCdtime() {
	return cdtime;
    }

    public void setCdtime(int cdtime) {
	this.cdtime = cdtime;
    }

    public String getDesc() {
	return desc;
    }

    public void setDesc(String desc) {
	this.desc = desc;
    }

    public String getSpellDesc() {
	return spellDesc;
    }

    public void setSpellDesc(String spellDesc) {
	this.spellDesc = spellDesc;
    }

    public int getMaxLevel() {
	return maxLevel;
    }

    public void setMaxLevel(int maxLevel) {
	this.maxLevel = maxLevel;
    }

    public int getChapter() {
	return chapter;
    }

    public void setChapter(int chapter) {
	this.chapter = chapter;
    }

    public int getPrice() {
	return price;
    }

    public void setPrice(int price) {
	this.price = price;
    }

    public void setReduceCdtimeLevelUp(float reduceCdtimeLevelUp) {
	this.reduceCdtimeLevelUp = reduceCdtimeLevelUp;
    }

    public float getReduceCdtimeLevelUp() {
	return reduceCdtimeLevelUp;
    }

    public String getSpell() {
	return spell;
    }

    public void setSpell(String spell) {
	this.spell = spell;
    }

    public int getNeedFragment() {
        return needFragment;
    }

    public void setNeedFragment(int needFragment) {
        this.needFragment = needFragment;
    }

}
