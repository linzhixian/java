package com.funugame.parkour.api.jsonbeans;

import java.io.Serializable;

import com.lzx.framework.web.springjson.beans.ResponseBase;

public class SubmitGameResultResponse extends ResponseBase implements Serializable {

    private static final long serialVersionUID = 1;

    private boolean win;
    private int type;
    private boolean hasFinishAchievement;

    public boolean isWin() {
	return win;
    }

    public void setWin(boolean win) {
	this.win = win;
    }

    public int getType() {
	return type;
    }

    public void setType(int type) {
	this.type = type;
    }

    public boolean isHasFinishAchievement() {
	return hasFinishAchievement;
    }

    public void setHasFinishAchievement(boolean hasFinishAchievement) {
	this.hasFinishAchievement = hasFinishAchievement;
    }
}
